create table;

create table ALUNOS(ID int not null primary key auto_increment, nome varchar(255) not null); (o comando auto_increment faz com que se crie um identificador unico de forma automática dentro da tabela);

insert into alunos (ID, nome) values (1, "Felipe");

update alunos set nome = "Leonardo" where ID = 2;

insert into alunos (ID, nome) values(null, "Guilherme");

insert into alunos(nome) values("Maria");

insert into alunos(ID, nome) values(75, "Lucas); \\quando se "quebra" o auto_increment, ele vai colocar a sequência de IDs a partir do maior número.

insert into alunos(nome) values ("Jordana");


alter table alunos add column ativo boolean not null default (true); \\ por padrão, será inserido uma coluna com "verdadeiro" ao lado de cada ID com o valor 1.

insert into alunos(nome, ativo) values ("Valentina", false); \\ aqui estou inserindo o valor "falso";

insert into alunos(nome) values ("Pamela"); \\ aqui será inserido o valor "1" pois o padrão é verdadeiro.

alter table alunos add column RA varchar (100) not null default ("SEM REGISTRO");

update alunos set RA = "123" where id = 1;
update alunos set RA = "456" where id = 2;
update alunos set RA = "789" where id = 3;
update alunos set RA = "101" where id = 4;

insert alunos (nome) values ("Loreni");

select *from alunos order by nome; \\ ordena os resultados em ordem alfabética na ordem normal.


select *from alunos order by nome desc; \\ ordena os resultados em ordem alfabética "de trás para frente".

select *from alunos where nome like "%G%;" // encontra no banco de dados qualquer nome que possua a letra G;
select *from alunos where nome like "G%" // encontra no banco de dados qualquer nome que inicia com a letra G;
select *from alunos where nome like "%G" // encontra no banco de dados qualquer nome que termina com a letra G;

select *from alunos where nome like "B% AND ativo = 1;
select *from alunos where nome like "B% OR nome like "%g%";
select *from alunos not where nome like "%u%; // seleciona qualquer nome que não possua a letra u.

alter table alunos add column data_nsc date; // adiciona a coluna de data de nascimento.
insert into aluno (nome, data_nsc) values ("Gabriel", "2001-03-25"); // para passar valores de data, colocar no formato aaaa-mm-dd.

select *from alunos order by data_nsc asc; // ordena do mais velho para o mais novo.
select *from alunos where data_nsc is not null order by data_nsc asc; // seleciona alunos cuja data de nascimento nao é nula e ordena por ordem do mais velho para o mais novo.
